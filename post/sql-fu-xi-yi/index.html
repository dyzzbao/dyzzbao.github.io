1 创建表

表结构如下：

生成的SQL文件为：

-- 学生表
DROP TABLE IF EXISTS `student`;
CREATE TABLE `student` (
  `学号` varchar(255) NOT NULL,
  `姓名` varchar(255) NOT NULL,
  `出生日期` date NOT NULL,
  `性别` varchar(255) NOT NULL,
  PRIMARY KEY (`学号`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
-- 成绩表
DROP TABLE IF EXISTS `score`;
CREATE TABLE `score` (
  `学号` varchar(255) NOT NULL,
  `课程号` varchar(255) NOT NULL,
  `成绩` float(3,0) NOT NULL,
  PRIMARY KEY (`学号`,`课程号`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
-- 课程表
DROP TABLE IF EXISTS `course`;
CREATE TABLE `course` (
  `课程号` varchar(255) NOT NULL,
  `课程名称` varchar(255) NOT NULL,
  `教师号` varchar(255) NOT NULL,
  PRIMARY KEY (`课程号`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
--  教师表
DROP TABLE IF EXISTS `teacher`;
CREATE TABLE `teacher` (
  `教师号` varchar(20) NOT NULL,
  `教师姓名` varchar(20) DEFAULT NULL,
  PRIMARY KEY (`教师号`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;

3 查询

-- 1. * 查询'0001'课程比'0002'课程成绩高的所有学生的学号
SELECT a.学号 FROM
(SELECT 学号,成绩 FROM score WHERE 课程号='0001') a
INNER JOIN (SELECT 学号,成绩 FROM score WHERE 课程号='0002') b
ON a.学号=b.学号
WHERE a.成绩>b.成绩;
-- 2. * 查询平均成绩大于60分学生的学号和平均成绩
SELECT 学号,AVG(成绩) AS 平均成绩 FROM score GROUP BY 学号 HAVING AVG(成绩) > 60;
-- 3. 查询所有学生的学号、姓名、选课数、总成绩
SELECT a.学号,a.姓名,COUNT(b.课程号) AS 选课数,SUM(b.成绩) AS 总成绩
FROM student a LEFT JOIN score b
ON a.学号=b.学号
GROUP BY a.学号,a.姓名;
-- 4. 查询姓“孟”老师的个数
SELECT COUNT(教师号) AS 总数 FROM teacher WHERE 教师姓名 LIKE '孟%';
-- 5. * 查询没学过“李辉”老师课的学生的学号、姓名
SELECT 学号,姓名
FROM student
WHERE 学号 NOT IN (
SELECT a.学号 FROM score a
INNER JOIN course b ON a.课程号=b.课程号
INNER JOIN teacher c ON b.教师号=c.教师号
WHERE c.教师姓名='李辉');
-- 6. * 查询学过“李辉”老师所教的所有课的同学的学号、姓名
SELECT 学号,姓名
FROM student
WHERE 学号 IN (
SELECT a.学号 FROM score a
INNER JOIN course b ON a.课程号=b.课程号
INNER JOIN teacher c ON b.教师号=c.教师号
WHERE c.教师姓名='李辉');
-- 7. * 查询学过编号为“0001”的课程并且也学过编号为“0002”的课程的学生的学号、姓名
SELECT 学号,姓名
FROM student
WHERE 学号 IN(
SELECT a.学号 FROM
(SELECT 学号,课程号 FROM score WHERE 课程号='0001') a
INNER JOIN (SELECT 学号,课程号 FROM score WHERE 课程号='0002') b
ON a.学号= b.学号);
-- 8. 查询课程编号为“0002”的总成绩
SELECT SUM(成绩) AS 总成绩 FROM score WHERE 课程号 = '0002';
-- 9. * 查询所有课程成绩小于60分学生的学号、姓名
SELECT 学号,姓名 FROM student WHERE 学号 IN
(SELECT 学号 FROM score WHERE 成绩< 60);
-- 10. * 查询没有学全所有课的学生的学号、姓名
SELECT 学号,姓名 FROM student WHERE 学号 IN (
SELECT 学号 FROM score GROUP BY 学号 HAVING COUNT(课程号)<
(SELECT COUNT(课程号) FROM course));
-- 11. * 查询至少有一门课与学号为“0005”的学生所学课程相同的学生的学号和姓名
SELECT DISTINCT a.学号,a.姓名
FROM student a INNER JOIN score b ON a.学号 = b.学号
WHERE b.课程号 IN
(SELECT 课程号 FROM score WHERE 学号 = '0005') AND a.学号!='0005';
-- 13. * 查询没学过"李辉"老师讲授的任一门课程的学号、姓名
SELECT 学号,姓名
FROM student
WHERE 学号 NOT IN
(SELECT a.学号
FROM score a
INNER JOIN course b ON a.课程号=b.课程号
INNER JOIN teacher c ON b.教师号=c.教师号
WHERE c.教师姓名='李辉');
-- 15. * 查询两门及以上不及格课程的同学的学号及其平均成绩
SELECT 学号, AVG(成绩) AS 平均成绩
FROM score
WHERE 成绩 < 60
GROUP BY 学号
HAVING COUNT(课程号) >= 2;
-- 16. * 检索课程编号为“0001”且分数小于60的学生学号，结果按分数降序排列
SELECT 学号,成绩 FROM score WHERE 课程号='0001' and 成绩 < 60 ORDER BY 成绩 DESC;
-- 17. * 按平均成绩从高到低显示所有学生的所有课程的成绩以及平均成绩
SELECT a.学号,a.课程号,a.成绩,b.平均成绩 FROM score a
INNER JOIN (SELECT 学号,AVG(成绩) AS 平均成绩 FROM score GROUP BY 学号) b
ON a.学号=b.学号
ORDER BY b.平均成绩 DESC
